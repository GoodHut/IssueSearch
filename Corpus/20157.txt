 URL : "https://github.com/riot/compiler/issues/100" TITLE : buble import error BODY : <custom-tag> <script type= buble > import 'my-file' </script> </custom-tag> results in: import is not supported error if you pass in the options {transforms: {modules: false}} into the buble compiler, the issue does not occur. see https://gitlab.com/rich-harris/buble/blob/master/src/program/types/importdeclaration.js here is a monkey-patch i used to fix the issue in my project. const parsers__riot = require 'riot-compiler/lib/parsers' , {js} = parsers__riot , {buble} = js js 'buble' = _loadparser js 'ecmascript-6' = _loadparser function _loadparser p1, p2, p3, p4 { return buble p1, object.assign p2, {transforms: {modules: false}} , p3, p4 }
